% Switch fonts according to language!
% Adapted from http://tex.stackexchange.com/a/129032/11851

\usepackage{polyglossia}
\usepackage[Latin, Khmer, Thai, Arabic]{ucharclasses}

\setDefaultTransitions{\normalfont}{}
\setTransitionsFor{Khmer}{\fontspec{Khmer}}{}  % From http://www.google.com/fonts/specimen/Khmer
\setTransitionsFor{Thai}{\fontspec{Thonburi}}{}
\setTransitionsFor{Arabic}{\fontspec{Al Tarikh}}{}
